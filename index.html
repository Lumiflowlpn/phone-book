function doGet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheets()[0]; const values = sheet.getDataRange().getValues(); if (values.length <= 1) return createJsonResponse([]); const headers = values[0]; const jsonData = values.slice(1).map(row => { const obj = {}; headers.forEach((header, i) => obj[header] = row[i]); return obj; }); return createJsonResponse(jsonData); } // 🆕 ฟังก์ชันรับข้อมูลแก้ไขจากหน้าเว็บเพื่อไป Update แถวที่ตรงกับ ID_Contact function doPost(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheets()[0]; const data = JSON.parse(e.postData.contents); const values = sheet.getDataRange().getValues(); const headers = values[0]; const idIndex = headers.indexOf("ID_Contact"); let rowIndex = -1; // ค้นหาแถวที่มี ID_Contact ตรงกับที่ส่งมา for (let i = 1; i < values.length; i++) { if (values[i][idIndex] == data.ID_Contact) { rowIndex = i + 1; // ลำดับแถวใน Sheets เริ่มที่ 1 break; } } if (rowIndex !== -1) { // ข้อมูลที่ต้องการอัปเดต (เรียงตามหัวตารางในชีตของคุณ) const updatedRow = [ data.ID_Contact, data["ชื่อ-สกุล"], data["ชื่อเล่น"], data["เบอร์โทรศัพท์"], data["ตำแหน่ง"], data["สังกัด"], data["หน่วยงาน/สถานศึกษา"], data["เบอร์โทรสถานศึกษา"] ]; // อัปเดตข้อมูลทั้งแถว sheet.getRange(rowIndex, 1, 1, updatedRow.length).setValues([updatedRow]); return createJsonResponse({ status: "success", message: "อัปเดตข้อมูลสำเร็จ" }); } else { return createJsonResponse({ status: "error", message: "ไม่พบรหัสบุคลากรนี้" }); } } catch (error) { return createJsonResponse({ status: "error", message: error.toString() }); } } function createJsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
